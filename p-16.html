Practical-16 : This Keyword 


class abc
{
    int a, b;

    void sum(int a, int b)
    {
        this.a = a;
        this.b = b;
        System.out.println("Sum is " + (a + b));
    }

    void display()
    {
        System.out.println("Mul is " + (a * b));
    }
}

class demo
{
    public static void main(String args[])
    {
        abc aa = new abc();
        aa.sum(10, 20);
        aa.display();
    }
}

Output:

Sum is 30
Mul is 200
