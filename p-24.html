Practical-24 : Multilevel Inheritance 


class abc
{
    void hello(int a, int b)
    {
        System.out.println("Ans = " + (a + b));  // Addition
    }
}

class xyz extends abc
{
    void display(int a, int b)
    {
        System.out.println("Ans = " + (a * b));  // Multiplication
    }
}

class pqr extends xyz
{
    void sum(int a, int b)
    {
        System.out.println("Ans = " + (a - b));  // Subtraction
    }
}

class demo
{
    public static void main(String args[])
    {
        pqr p = new pqr();  // Creating object of the child class
        p.hello(16, 4);     // Accessing method from 'abc' class (addition)
        p.display(4, 16);   // Accessing method from 'xyz' class (multiplication)
        p.sum(10, 5);       // Accessing method from 'pqr' class (subtraction)
    }
}

Output:

Ans = 20
Ans = 64
Ans = 5
