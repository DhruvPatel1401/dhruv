Practical-18 : Array List


import java.util.ArrayList;

class demo
{
    public static void main(String args[])
    {
        ArrayList stu = new ArrayList();
        stu.add("Hi");
        stu.add("Dhruv");
        stu.add("Patel");
        System.out.println("Ans:" + stu);
    }
}

Output:

Ans: [Hi, Dhruv, Patel]
