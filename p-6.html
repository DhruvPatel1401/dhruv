Practical-6 : Method Overloading


class xyz
{
    void sum(int a, int b)
    {
        System.out.println("Ans-1 = " + (a + b));
    }

    void sum(int a)
    {
        System.out.println("Ans-2 = " + (a * 2));
    }

    void sum()
    {
        System.out.println("Ans-3 = " + (2 + 2));
    }
}

class demo
{
    public static void main(String[] args)
    {
        xyz x = new xyz();
        x.sum(8, 4);
        x.sum(4);
        x.sum();
    }
}

Output:

Ans-1 = 12
Ans-2 = 8
Ans-3 = 4
