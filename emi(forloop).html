EMI Calculation Using For Loop


import java.util.Scanner;
import java.lang.Math;

class abc
{
    void emi(double p, double r, double n)
    {
        double emi;
        emi = (p * (r / 1200) * Math.pow((1 + (r / 1200)), (n * 12))) / (Math.pow((1 + (r / 1200)), (n * 12)) - 1);
        int emii = (int) emi;
        System.out.println("EMI IS:" + emi);
        System.out.print("\n");
        for (int i = 1; i <= n * 12; i++)
        {
            double iamt = p * (r / 1200);
            int a = (int) iamt;
            double pamt = emi - iamt;
            int b = (int) pamt;
            double rem = p - pamt;
            int c = (int) rem;
            System.out.println(i + "||" + b + "||" + a + "||" + emi + "\\" + "c" + "\n");
            p = rem;
        }
    }
}

class demo
{
    public static void main(String args[])
    {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter Loan Amt:");
        double p = sc.nextDouble();
        System.out.print("Enter ROI:");
        double r = sc.nextDouble();
        System.out.print("Enter Year:");
        double n = sc.nextDouble();
        System.out.print("\n");
        abc aa = new abc();
        aa.emi(p, r, n);
    }
}
