Practical-12 : For Loop


class abc
{
    void cal(int a)
    {
        for (int i = 1; i <= a; i++)
        {
            System.out.println("Ans: " + i);
        }
    }
}

class demo
{
    public static void main(String[] args)
    {
        abc aa = new abc();
        aa.cal(10);
    }
}

Output:

Ans: 1 Ans: 2 Ans: 3 Ans: 4 Ans: 5 Ans: 6 Ans: 7 Ans: 8 Ans: 9 Ans: 10
