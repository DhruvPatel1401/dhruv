CAGR Calculation - Java Code


import java.util.Scanner;
import java.lang.Math;

class demo
{
    public static void main(String args[]) 
    {
        Scanner sc = new Scanner(System.in);
        double[][] a = new double[2][8];
        int i;
        System.out.print("Enter Amount: ");
        double aa = sc.nextDouble();
        
        // Get percentage inputs for 5 years
        for(i = 0; i <= 4; i++)
        {
            System.out.print("Enter Percentage: ");
            a[1][i] = sc.nextDouble();
        }
        
        // First year calculation
        a[0][0] = (aa * (a[1][0] / 100)) + aa;
        
        // Calculate subsequent years
        for(i = 0; i <= 4; i++)
        {
            a[0][i+1] = (a[0][i] * (a[1][i+1] / 100)) + a[0][i];
        }
        
        // Display results for all years
        for(i = 0; i <= 4; i++)
        {
            System.out.print("Ans: " + a[0][i] + "\n");
        }
        
        // Calculate CAGR
        double n = 5;
        double cagr = (Math.pow((a[0][4] / aa), (1 / n)) - 1) * 100;
        System.out.print("CAGR Is: " + cagr);
    }
}
