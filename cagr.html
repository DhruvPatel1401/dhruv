import java.util.Scanner;

class abc
{
    public static void main(String[] args)
    {
        Scanner scanner = new Scanner(System.in);

        float aa;
        float[][] a = new float[2][4];
        float opb, clb, n, cagr;

        System.out.print("Enter your value: ");
        aa = scanner.nextFloat();

        for (int i = 0; i < 4; i++)
        {
            System.out.print("Enter percentage: ");
            a[1][i] = scanner.nextFloat();
        }

        a[0][0] = (aa * (a[1][0] / 100)) + aa;
        for (int i = 0; i < 3; i++)
        {
            a[0][i + 1] = (a[0][i] * (a[1][i + 1] / 100)) + a[0][i];
        }

        for (int i = 0; i < 4; i++)
        {
            System.out.printf("Answer is=%.2f\n", a[0][i]);
        }
    }
}
