Practical-4 : Multiple Class


class abc
{
    void sum(int a, int b)
    {
        System.out.println("Sum of=" + (a + b));
    }

    void mul(int a, int b)
    {
        System.out.println("Mul of=" + (a * b));
    }
}

class xyz
{
    void cal(int a, int b)
    {
        System.out.println("Cal of=" + ((a + b) * 2));
    }
}

class demo
{
    public static void main(String args[])
    {
        abc a = new abc();
        xyz x = new xyz();
        a.sum(2, 3);
        a.mul(3, 4);
        x.cal(3, 5);
    }
}

Output:

Sum of=5
Mul of=12
Cal of=16
