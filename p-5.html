Practical-5 : Multiple Method


class abc
{
    int x, y;

    void sum(int a, int b)
    {
        x = a;
        y = b;
        System.out.println("Sum of=" + (a + b));
    }

    void mul()
    {
        System.out.println("Mul of=" + (x * y));
    }
}

class demo
{
    public static void main(String args[])
    {
        abc aa = new abc();
        int a, b;
        a = 2;
        b = 3;
        aa.sum(a, b);
        aa.mul();
    }
}

Output:

Sum of=5
Mul of=6
