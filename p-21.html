Practical-21 : Wrapper Class- 2


class demo
{
    public static void main(String args[])
    {
        Integer a = 5;
        Double b = 6.6;
        Character c = 'A';
        
        System.out.println("Ans = " + a.intValue());
        System.out.println("Ans = " + b.doubleValue());
        System.out.println("Ans = " + c.charValue());
    }
}

Output:

Ans = 5
Ans = 6.6
Ans = A
