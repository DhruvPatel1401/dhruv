Practical-14 : EMI Calculation 


import java.util.Scanner;

class abc
{
    void emi(double p, double r, double n)
    {
        double ans;
        ans = (p * (r / 1200) * Math.pow((1 + (r / 1200)), (n * 12))) / (Math.pow((1 + (r / 1200)), (n * 12)) - 1);
        int anss = (int) ans;
        System.out.println("EMI = " + anss);
    }
}

public class demo
{
    public static void main(String args[])
    {
        abc aa = new abc();
        Scanner sc = new Scanner(System.in);

        System.out.print("Loan Amt = ");
        double p = sc.nextDouble();

        System.out.print("ROI = ");
        double r = sc.nextDouble();

        System.out.print("Year = ");
        double n = sc.nextDouble();

        aa.emi(p, r, n);
    }
}
