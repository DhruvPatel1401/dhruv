EMI Calculation with Float Value


import java.util.Scanner;

class abc
{
    void emi(double p, double r, double n)
    {
        double ans;
        ans = (p * r * Math.pow((1 + (r / 1200)), (n * 12))) / (Math.pow((1 + (r / 1200)), (n * 12)) - 1);
        //int anss = (int)ans;
        System.out.println("EMI=" + ans);
    }
}

class demo
{
    public static void main(String args[])
    {
        abc aa = new abc();
        aa.emi(1200000, 10, 1);
    }
}

Output:

EMI=104697.442
