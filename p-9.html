Practical-9 : Polymorphism


class xyz
{
    int x, y;
    void sum(int a, int b)
    {
        x = a;
        y = b;
        System.out.println("Ans = " + (a + b));
    }

    void sum(int a)
    {
        System.out.println("Ans = " + (a * y));
    }

    void sum()
    {
        System.out.println("Ans = " + (x / y));
    }
}

class demo
{
    public static void main(String[] args)
    {
        xyz x = new xyz();
        x.sum(8, 4);
        x.sum(4);
        x.sum();
    }
}

Output:

Ans = 12 Ans = 16 Ans = 2
