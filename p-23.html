Practical-23 : Exception Handling / Try Catch - 2


class abc
{
    int a, b;
    void sum(int a, int b)
    {
        this.a = a;
        this.b = b;
        int ans;
        try
        {
            ans = a / b;
            System.out.println("Ans = " + ans);
        }
        catch (Exception e)
        {
            System.out.println("Error: Divide by zero");
        }
    }

    void display()
    {
        System.out.println("Sum = " + (a + b));
    }
}

class demo
{
    public static void main(String args[])
    {
        abc aa = new abc();
        aa.sum(2, 8);  // No exception, division succeeds
        aa.display();  // Prints sum of a and b
    }
}

Output:

Ans = 0
Sum = 10
