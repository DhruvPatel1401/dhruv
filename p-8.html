Practical-8 : Single Inheritance


class abc
{
    void sum(int a, int b)
    {
        System.out.println("Ans = " + (a + b));
    }
}

class xyz extends abc
{
    void mul(int a, int b)
    {
        System.out.println("Ans = " + (a * b));        
    }
}

class demo
{
    public static void main(String[] args)
    {
        xyz x = new xyz();
        x.sum(8, 4);
        x.mul(7, 8);
    }
}

Output:

Ans = 12 Ans = 56
