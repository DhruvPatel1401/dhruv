Practical-19 : ArrayList Methods(Add,Set,Get,Remove,Size,Clear) 


import java.util.ArrayList;

class demo
{
    public static void main(String args[])
    {
        ArrayList stu = new ArrayList();
        
        // Add elements
        stu.add("Hi");
        stu.add("Dhruv");
        stu.add("Patel");
        System.out.println("Ans: " + stu);
        
        // Add element at specific index
        stu.add(1, "Good Morning");
        System.out.println("Ans: " + stu);
        
        // Get element from a specific index
        System.out.println("Ans: " + stu.get(2));
        
        // Set element at specific index
        stu.set(3, "BBB");
        System.out.println("Ans: " + stu);
        
        // Remove element from specific index
        stu.remove(1);
        System.out.println("Ans: " + stu);
        
        // Get the size of the ArrayList
        System.out.println("SIZE of: " + stu.size());
        
        // Clear all elements
        stu.clear();
        System.out.println("STU DATA: " + stu);
    }
}

Output:

Ans: [Hi, Dhruv, Patel]
Ans: [Hi, Good Morning, Dhruv, Patel]
Ans: Dhruv
Ans: [Hi, Good Morning, Dhruv, BBB]
Ans: [Hi, Dhruv, BBB]
SIZE of: 3
STU DATA: []
