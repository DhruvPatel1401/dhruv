Practical-11 : Constructor Overloading


class abc
{
    abc(int a, int b)
    {
        System.out.println("Ans = " + (a + b));
    }

    abc(int a, int b, int c)
    {
        System.out.println("Ans = " + (a + b + c));
    }

    abc()
    {
        System.out.println("Ans = " + (2 + 3));
    }
}

class demo
{
    public static void main(String[] args)
    {
        abc aa1 = new abc(2, 3);
        abc aa2 = new abc(3, 4, 5);
        abc a = new abc();
    }
}

Output:

Ans = 5 Ans = 12 Ans = 5
